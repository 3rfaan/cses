# NUMBER SPIRAL

| var   | desc                     |
| ----- | ------------------------ |
| $t$   | No. of tests             |
| $y,x$ | Row, column              |
| $r,c$ | $r \equiv y, c \equiv x$ |

## r == c

`(1,1) -> 1 * 1 = 1 - 0 = 1`

`(2,2) -> 2 * 2 = 4 - 1 = 3`

`(3,3) -> 3 * 3 = 9 - 2 = 7`

### r

$r \times r - (r - 1)$

$r^2 - r + 1$

### c

$c \times c - (c - 1)$

$c^2 - c + 1$

## r > c

### r % 2 == 0

`(4,1) -> 4 * 4 = 16 - 0 = 16`

`(4,2) -> 4 * 4 = 16 - 1 = 15`

`(4,3) -> 4 * 4 = 16 - 2 = 14`

$r^2 - (c - 1)$

$r^2 - c + 1$

### r % 2 != 0

`(5,1) -> 5 * 5 = 25 - 8 = 17`

`(5,2) -> 5 * 5 = 25 - 7 = 18`

`(5,3) -> 5 * 5 = 25 - 6 = 19`

`(5,4) -> 5 * 5 = 25 - 5 = 20`

$r^2 - r + 1 - (r - c)$

$r^2 - r + 1 - r + c$

$r^2 - 2r + 1 + c$

$(r - 1)^2 + c$

## c > r

### c % 2 == 0

`(1,4) -> 4 * 4 = 16 - 6 = 10`

`(2,4) -> 4 * 4 = 16 - 5 = 11`

`(3,4) -> 4 * 4 = 16 - 4 = 12`

$c^2 - c + 1 - (c - r)$

$c^2 - c + 1 - c + r$

$c^2 - 2c + 1 + r$

$(c - 1)^2 + r$

### c % 2 != 0

`(1,5) -> 5 * 5 = 25`

`(2,5) -> 5 * 5 = 25 - 1 = 24`

`(3,5) -> 5 * 5 = 25 - 2 = 23`

`(4,5) -> 5 * 5 = 25 - 3 = 22`

$c^2 - c + 1 + c - r$

$c^2 - r + 1$
